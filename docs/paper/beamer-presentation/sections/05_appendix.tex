\begin{frame}{A1: Homophenes --- The Lip-Reading Problem}
  \twocol{0.52}{
    {\small 50--70\% of English sounds are \alert{invisible} on lips.\\
    Multiple sounds produce identical mouth shapes:}

    \vspace{6pt}

    \begin{tabular}{@{}ll@{}}
      \toprule
      \textbf{Viseme Group} & \textbf{Sounds} \\
      \midrule
      Bilabial & p, b, m \\
      Alveolar & t, d, n, s, z, l \\
      Velar & k, g, ng \\
      Labiodental & f, v \\
      \bottomrule
    \end{tabular}

    \vspace{6pt}
    \notetext{``Viseme'' = visual phoneme. One viseme maps\\to multiple phonemes.}
  }{0.45}{
    {\small\textbf{Confusable word pairs:}}

    \vspace{4pt}

    \begin{tabular}{@{}l@{$\;\leftrightarrow\;$}l@{}}
      \toprule
      \multicolumn{2}{c}{\textbf{Identical on Lips}} \\
      \midrule
      mom & bomb \\
      pat & bat \\
      collar & color \\
      pads & pants \\
      admiral & animal \\
      probiotics & permafrost \\
      \bottomrule
    \end{tabular}

    \vspace{6pt}
    \notetext{Context is the ONLY disambiguation signal.\\This is why the LLM component matters.}
  }

  \note{
    Q\&A backup for ``why is lip-reading so hard?''
    50--70\% of sounds invisible on lips --- same mouth shape for multiple phonemes.
    ``admiral'' vs ``animal'': identical bilabial + alveolar sequence.
    Context (the LLM) is the ONLY disambiguation signal.
  }
\end{frame}

\begin{frame}[shrink=10]{A14: Metric Mismatch Guide}
  {\small ``When you see X, it means Y'' --- 8 diagnostic patterns:}

  \vspace{4pt}

  {\scriptsize
  \begin{tabular}{@{}p{3.5cm}p{5cm}p{3cm}@{}}
    \toprule
    \textbf{Pattern} & \textbf{What It Means} & \textbf{IS Signal} \\
    \midrule
    \alert{WER $\gg$ WWER} (gap $\geq$ 10pp) &
    Function words wrong, content words right. Meaning likely preserved. &
    Inv-WWER catches it \\[3pt]
    WWER $\gg$ WER (rare) &
    Content words wrong, function words right. \alert{Worse than it looks.} &
    Inv-WWER penalizes \\[3pt]
    High Semantic + High WER (sem $\geq$ 0.5, WER $\geq$ 60) &
    Paraphrasing, not error. Different words, same meaning. &
    Semantic preserves credit \\[3pt]
    Low Semantic + Low WER (sem $<$ 0.3, WER $<$ 40) &
    \alert{Entity destruction} --- few words changed but meaning lost. &
    Semantic penalizes \\[3pt]
    High Phonetic + Low Semantic (phon $\geq$ 0.7, sem $<$ 0.3) &
    Homophene errors --- sounds right, means wrong. Classic lip-reading. &
    Phonetic vs Semantic \\[3pt]
    High NEA + High WWER (NEA $\geq$ 50, WWER $\geq$ 60) &
    Names/numbers correct, everything else wrong. Key facts survive. &
    NEA F1 preserves credit \\[3pt]
    Length Ratio $>$ 1.5 + High WER &
    Over-generation / hallucination --- model added fabricated text. &
    Length Ratio penalizes \\[3pt]
    Length Ratio $<$ 0.5 + Low WER &
    Truncation --- partial capture, stopped too early. &
    Length Ratio penalizes \\
    \bottomrule
  \end{tabular}}

  \vspace{4pt}
  \notetext{Each mismatch pattern reveals a different system behavior. Multiple patterns can co-occur in one segment.}

  \note{
    Q\&A backup for ``why not just use WER?''
    8 diagnostic patterns show how different metric combinations reveal different system behaviors.
    Key example: WER $\gg$ WWER means function words wrong but meaning preserved --- WER over-penalizes.
    Low Semantic + Low WER = entity destruction --- few words changed but meaning lost entirely.
    See A14b for concrete examples of each pattern.
  }
\end{frame}
