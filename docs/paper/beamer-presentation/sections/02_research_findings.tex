\begin{frame}[shrink=15]{IS Deep Dive: What Drives Quality?}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{\small 6 Signals $\rightarrow$ 3 Independent Dimensions}
        {\scriptsize
        \begin{tabular}{@{}lcc@{}}
          \textbf{Signal} & \textbf{r with IS} & \textbf{Variance} \\
          \midrule
          Phonetic Sim & \keyword{0.943} & 14.2\% \\
          Semantic Sim & 0.921 & 28.5\% \\
          WWER & $-$0.881 & 15.2\% \\
          NEA F1 & 0.864 & 17.3\% \\
          WER & $-$0.849 & 15.7\% \\
          Length Ratio & 0.226 & 9.1\% \\
        \end{tabular}\\[4pt]
        \textbf{Dimensions:}\\
        1. \keyword{Word accuracy} (WER, WWER, Phonetic) --- 60\%\\
        2. \keyword{Meaning} (Semantic) --- 28\%\\
        3. \keyword{Output sanity} (Length Ratio) --- 9\%}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{\small Claude-as-a-Judge Methodology}
        {\scriptsize
        \textbf{Design time:} Claude (Anthropic) acted as\\
        expert judge to design the IS framework:\\[2pt]
        \begin{itemize}\scriptsize
          \item Selected 6 linguistic signals
          \item Assigned weights (semantic 0.25, rest 0.15)
          \item Defined 5-tier quality boundaries
          \item Validated against human intuition
        \end{itemize}
        \vspace{2pt}
        \textbf{Run time:} Fully deterministic ---\\
        no LLM calls needed per sample.\\[2pt]
        \textbf{Validation:} \keyword{r = 0.925} across 16 configs,\\
        $\kappa$ = 0.77, recall 97.6--100\%.\\
        Strictly additive: $R^2 = 1.000$.}
      \end{block}
    \end{column}
  \end{columns}

  \vspace{4pt}
  \centering
  {\scriptsize\textbf{Formula:} IS = 0.25$\times$Semantic + 0.15$\times$(Phonetic + InvWER + InvWWER + NEA + LengthRatio)}

  \note{
    KEY CORRELATION FINDINGS:
    1. Phonetic Sim: strongest single predictor (r=0.943) despite 15\% weight ---
    most direct measure of visual encoder quality.
    2. 4 of 6 signals measure the same thing (word accuracy cluster, r > 0.79) ---
    60\% of IS is ``did the model get the words right?''
    3. Semantic Sim (25\% weight) drives most IS variance (28.5\%) ---
    the tiebreaker signal separating segments with similar word accuracy.
    4. NEA F1 punches above its weight: 17.3\% variance from 15\% weight.
    Names/numbers are binary --- either preserved or not.
    5. WER is UNRELIABLE across configs: correlation swings from -0.95 to -0.45
    with length penalty changes. This is why IS was created over raw WER.
    6. Length Ratio nearly useless (9.1\%, sign flips across configs).
    7. Per-tier: Failed tier dominated by Phonetic (``does it sound right?''),
    Excellent tier dominated by WER (``exact word accuracy'').
    Claude designed the entire framework at design time. At run time: fully
    deterministic, free, reproducible. R-squared = 1.0.
    Cross-config: r=0.925 mean, recall 97.6--100\% across 16 decode configs.
    See A14 for Metric Mismatch Guide --- 8 diagnostic patterns for interpreting signal combinations.
    See A11 for LLM Salvage analysis of segments where metrics understate true quality.
  }
\end{frame}

\begin{frame}[shrink=8]{Why the Gap: Root Causes}
  \twocol{0.48}{
    \textbf{Three root causes:}
    \begin{enumerate}
      \item \alert{Domain mismatch}\\
            {\scriptsize TED talks $\neq$ YouTube}
      \item \alert{Hallucination}\\
            {\scriptsize LLM prior overwhelms visual signal}
      \item \alert{Short-segment failure}\\
            {\scriptsize Not enough visual context}
    \end{enumerate}

    \vspace{6pt}

    \begin{exampleblock}{\small Topic Analysis}
      \begin{tabular}{@{}lcc@{}}
        \scriptsize\textbf{Topic} & \scriptsize\textbf{IS} & \scriptsize\textbf{Captured} \\

        \scriptsize DIY/Home & \scriptsize\tierpoor{2.13} & \scriptsize 30\% \\
      \end{tabular}

    \vspace{2pt}
    \notetext{Formal vocabulary $\rightarrow$ better predictions}

\begin{frame}[shrink=5]{Limits of Tuning}
  \twocol{0.48}{
    {\large\keyword{Tuning = mitigation, not cure}}

    \vspace{4pt}

    \begin{itemize}\small
      \item Config J: eliminates empties,\\doubles hallucinations
      \item Rankings \keyword{stable} (r $>$ 0.92)\\across all 16 configs
      \item ``Hard'' and ``easy'' segments\\stay the same
    \end{itemize}

    \vspace{4pt}

    \begin{alertblock}{\scriptsize Bottleneck: Visual Encoder}
      {\small Three levers remain:}
      \begin{enumerate}\scriptsize
        \item Scale data to 20K--50K segments
        \item Swap to stronger LLM
        \item Smart prompts
      \end{enumerate}
    \end{alertblock}
  }{0.49}{
    \halfwidthfig{P4_lenpen_sensitivity.png}

    \vspace{2pt}
    \notetext{Length penalty vs empty/hallucination trade-off}
  }

  \note{
    Tuning cannot fix the fundamental problem.
    Cross-config proof: segment rankings stable (r>0.92).
    The visual encoder determines which segments succeed or fail.
    Three remaining levers: data, LLM, and prompts.
    This sets up the entire Future Directions section.
  }
\end{frame}
