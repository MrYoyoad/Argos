\begin{frame}[shrink=8]{Why the Gap: Root Causes}
  \twocol{0.48}{
    \textbf{Three root causes:}
    \begin{enumerate}
      \item \alert{Domain mismatch}\\
            {\scriptsize TED talks $\neq$ YouTube}
      \item \alert{Hallucination}\\
            {\scriptsize LLM prior overwhelms visual signal}
      \item \alert{Short-segment failure}\\
            {\scriptsize Not enough visual context}
    \end{enumerate}

    \vspace{6pt}

    \begin{exampleblock}{\small Topic Analysis}
      \begin{tabular}{@{}lcc@{}}
        \scriptsize\textbf{Topic} & \scriptsize\textbf{IS} & \scriptsize\textbf{Captured} \\
        \scriptsize Business/Finance & \scriptsize\tiergood{3.08} & \scriptsize 57\% \\
        \scriptsize Education & \scriptsize\tierfair{2.67} & \scriptsize 43\% \\
        \scriptsize DIY/Home & \scriptsize\tierpoor{2.13} & \scriptsize 30\% \\
      \end{tabular}
    \end{exampleblock}

    \vspace{2pt}
    \notetext{Formal vocabulary $\rightarrow$ better predictions}
  }{0.49}{
    \halfwidthfig{01_wer_vs_duration.png}

    \vspace{2pt}
    \begin{center}
    {\small\textbf{Length matters:}}\\[2pt]
    \begin{tabular}{@{}lcc@{}}
      \scriptsize\textbf{Words} & \scriptsize\textbf{WER} & \scriptsize\textbf{Captured} \\
      \scriptsize 5--10 & \scriptsize\werbad{74\%} & \scriptsize 32\% \\
      \scriptsize 20+ & \scriptsize\werneutral{55\%} & \scriptsize 49\% \\
    \end{tabular}
    \end{center}
  }

  \note{
    Short segments catastrophic --- 5--10 words only 32\% captured vs 49\% for 20+.
    Business/Finance best at 57\% (formal vocabulary helps LLM).
    DIY/Home worst at 30\% (casual speech, technical jargon).
    Length is the single strongest predictor of success.
  }

% Slide 13: Limits of Tuning
% ----------------------------------------------------------
\begin{frame}{Limits of Tuning}
  \twocol{0.48}{
    {\large\keyword{Tuning = mitigation, not cure}}

    \vspace{4pt}

    \begin{itemize}\small
      \item Config J: eliminates empties,\\doubles hallucinations
      \item Rankings \keyword{stable} (r $>$ 0.92)\\across all 16 configs
      \item ``Hard'' and ``easy'' segments\\stay the same
    \end{itemize}

    \vspace{4pt}

    \begin{alertblock}{\scriptsize Bottleneck: Visual Encoder}
      {\small Three levers remain:}
      \begin{enumerate}\scriptsize
        \item Scale data to 20K--50K segments
        \item Swap to stronger LLM
        \item Smart prompts
      \end{enumerate}
    \end{alertblock}
  }{0.49}{
    \halfwidthfig{P4_lenpen_sensitivity.png}

    \vspace{2pt}
    \notetext{Length penalty vs empty/hallucination trade-off}
  }

  \note{
    Tuning cannot fix the fundamental problem.
    Cross-config proof: segment rankings stable (r>0.92).
    The visual encoder determines which segments succeed or fail.
    Three remaining levers: data, LLM, and prompts.
    This sets up the entire Future Directions section.
  }
\end{frame}
