% Word Error Rate: The Standard Metric
% ----------------------------------------------------------
\begin{frame}{Word Error Rate: The Standard Metric}
  \twocol{0.52}{
    \begin{block}{\small WER Formula}
      $$\text{WER} = \frac{S + I + D}{N} \times 100\%$$
      {\scriptsize where $S$ = substitutions, $I$ = insertions,\\
      $D$ = deletions, $N$ = reference word count}
    \end{block}

    \vspace{4pt}

    \begin{exampleblock}{\small Worked Example}
      {\scriptsize
      \textbf{Reference:} ``the cat sat on the mat'' (6 words)\\
      \textbf{Hypothesis:} ``a cat sat on a mat''\\[2pt]
      2 substitutions (the$\rightarrow$a), 0 insertions, 0 deletions\\
      WER = $\frac{2}{6}$ = \keyword{33\%}}
    \end{exampleblock}
  }{0.45}{
    \textbf{What WER captures:}
    \begin{itemize}\small
      \item Total word-level accuracy
      \item Industry standard for ASR
      \item Easy to compute and compare
    \end{itemize}

    \vspace{6pt}

    \textbf{\textcolor{ArgosRed}{What WER misses:}}
    \begin{itemize}\small
      \item \alert{Meaning preservation}
      \item Entity importance (names vs ``the'')
      \item Phonetic plausibility
      \item Output length pathologies
    \end{itemize}

    \vspace{4pt}
    \notetext{WER $<$ 30\% generally considered ``usable''\\in speech recognition.}
  }

  \note{
    Brief WER primer for non-specialists. WER is the standard metric in speech recognition.
    Simple formula: count word-level errors (substitutions, insertions, deletions) divided by
    reference length. WER under 30\% is generally considered ``usable.''
    BUT: WER treats all errors equally. Changing ``the'' to ``a'' counts the same as
    changing ``admiral'' to ``animal.'' This matters enormously for lip reading.
  }
\end{frame}

    \end{itemize}
  }

% Design Philosophy: Deterministic, Not AI

\begin{frame}{Design Philosophy: Deterministic, Not AI}
  \twocol{0.50}{
    \textbf{\textcolor{ArgosOrange}{Option A: LLM Per Sample}}

    \vspace{4pt}

    \begin{itemize}\small
      \item Call Claude/GPT-4 for each segment
      \item \alert{Expensive} (\$0.01--0.05 per sample)
      \item Non-reproducible (temperature, version drift)
      \item Slow (API latency per evaluation)
    \end{itemize}
  }{0.47}{
    \textbf{\textcolor{ArgosTeal}{Option B: Design-Time LLM (Our Approach)}}

    \vspace{4pt}

    \begin{itemize}\small
      \item Claude \textbf{designed the rubric} at development time
      \item Selected 6 signals, weights, tier boundaries
      \item Distilled into deterministic Python code
      \item \keyword{Runtime: pure math, \$0, instant}
      \item Reproducible, decomposable, auditable
    \end{itemize}
  }

  \vspace{6pt}
  \centering
  {\small Validated: \keyword{88.6\%} agreement with full LLM judge (r = 0.85 with blind evaluation).}

  \note{
    Two approaches to quality evaluation. We chose Option B: use Claude at design time
    to create the rubric, then encode it into deterministic code.
    Result: IS pipeline is pure Python math --- \$0 cost, runs in seconds, fully reproducible.
    Validated at 88.6\% agreement with full LLM-as-a-Judge evaluation (r = 0.85).
  }
\end{frame}

% IS: Six Signals of Quality
% ----------------------------------------------------------
\begin{frame}[shrink=5]{IS: Six Signals of Quality}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{\small Semantic Similarity (25\%)}
        {\scriptsize Do the sentences mean the same thing?\\
        Sentence embeddings (SBERT).\\[2pt]
        \keyword{Heaviest weight} --- meaning is the goal.}
      \end{block}

      \vspace{2pt}

      \begin{block}{\small Phonetic Similarity (15\%)}
        {\scriptsize Do they \textit{sound} alike?\\
        Phoneme-level edit distance.\\[2pt]
        \keyword{Strongest predictor} (r = 0.943 with IS).\\
        Natural for lip reading: visemes $\rightarrow$ phonemes.}
      \end{block}

      \vspace{2pt}

      \begin{block}{\small Inverse WER (15\%)}
        {\scriptsize How many words are correct?\\
        Standard word error rate, inverted ($1 - \text{WER}$).}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{\small Inverse WWER (15\%)}
        {\scriptsize Are \textit{content words} preserved?\\
        Content words weighted 2$\times$\\[2pt]
        ``admiral'' matters more than ``the''.}
      \end{block}

      \vspace{2pt}

      \begin{block}{\small Named Entity F1 (15\%)}
        {\scriptsize Are names, numbers, places preserved?\\
        \alert{Cannot be recovered from context}.\\[2pt]
        Largest differentiator: 74\% vs 16\% (58pp gap).}
      \end{block}

      \vspace{2pt}

      \begin{block}{\small Length Ratio (15\%)}
        {\scriptsize Is the output the right length?\\
        Catches hallucination (too long) and\\
        truncation (too short).}
      \end{block}
    \end{column}
  \end{columns}

  \vspace{4pt}
  \centering
  {\small\textbf{Formula:} IS = 0.25$\times$Semantic + 0.15$\times$(Phonetic + InvWER + InvWWER + NEA + LengthRatio)}\\
  {\small Score: 0--5 \quad Threshold: IS $\geq$ 3.0 = ``Properly Captured''}

  \note{
    Six signals explained step by step. Semantic similarity (25\%) is the heaviest ---
    meaning preservation is the whole point. Phonetic similarity (15\%) is the strongest
    single predictor (r=0.943) --- natural for lip reading where visual ambiguity maps to
    phonetic similarity. WER and WWER measure word accuracy at different granularities.
    NEA F1 captures names/numbers which can't be guessed from context --- the largest
    success/failure differentiator (58pp gap). Length ratio catches output pathologies.
    Formula at the bottom: weighted sum producing a 0--5 score.
  }
\end{frame}

  }{0.42}{
    \alertmetric{19\%}{Domain Vocabulary Confusion}

    \vspace{4pt}

    {\small$\sim$284 segments show systematic vocabulary confusion. \keyword{Topic-aware fine-tuning} is needed --- naive label injection at decode time does not help (tested, $-$0.9pp).}

    \vspace{6pt}

    {\scriptsize
    \textbf{Why Business/Finance works:}\\
    Formal vocabulary matches LLM training data.\\[4pt]
    \textbf{Why DIY/Home fails:}\\
    Casual speech, slang, technical jargon\\
    are alien to the TED-talk-trained model.}
  }

  \note{
    Full topic analysis showing why domain matters. Business/Finance at 57\% captured
    vs DIY/Home at 30\%. The LLM was trained on TED-talk-style formal speech.
    ~284 segments (19\%) show domain vocabulary confusion --- the model substitutes
    domain-appropriate words with generic TED-talk vocabulary.
    \\[6pt]
    \textbf{Where does the 284 come from?} The LLM-as-a-Judge evaluation classified
    failure modes for all 1,497 segments: 143 tagged ``Total Topic Drift'' (fluent text,
    completely wrong topic) + 141 tagged ``Phonetically Similar but Wrong Topic''
    (correct sounds, wrong domain vocabulary) = 284.
    \\[6pt]
    \textbf{How would topic labels technically work?} The VSP-LLM architecture prepends
    a text instruction to visual features before feeding them to Llama-2-7B. The token
    sequence is: [instruction tokens] [visual encoder features] [label tokens].
    A topic label would be injected at the \emph{beginning} of the instruction:
    ``The speaker is discussing \{Topic\}. Recognize this speech in English. Input~:~''
    --- this is a one-line code change in \texttt{vsp\_llm\_dataset.py}, no model
    retraining needed.
    \\[6pt]
    \textbf{Experiment (March 2026):} We ran this exact experiment on all 284 segments.
    Result: \textbf{topic labels do NOT help} without model fine-tuning.
    Baseline WER 86.6\% vs Topic WER 87.6\% ($-$0.9pp, worse).
    47 segments improved (17\%), 55 degraded (19\%), 182 unchanged (64\%).
    Critical failure: 67 segments (24\%) echoed ``the speaker is discussing technology''
    literally instead of recognizing speech. Cooking/Food labels made things 10.3pp worse
    (mislabeled segments). The model was trained with a fixed instruction format;
    extra tokens shift visual feature positions and break learned attention patterns.
    This is prompt engineering on a model not trained for it.
    \\[6pt]
    \textbf{Control group (100 good segments, IS$\geq$3):} Topic labels cause $-$0.6pp
    degradation (29.3\% $\to$ 30.0\%) but \emph{zero} instruction echoing. When visual
    signal is strong, the model ignores the extra tokens. Echoing only happens on weak segments.
    \\[6pt]
    \textbf{What would work:} Fine-tune with topic-prefixed instructions in training
    data so the model learns to use topic context. Requires training data augmentation,
    not just inference-time prompt changes.
  }
\end{frame}

\begin{frame}{What Are Decode Parameters?}
  \vspace{2pt}
  {\small Before showing our experiments --- a brief primer on what we can control:}

  \vspace{6pt}

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{\small Beam Size}
        {\scriptsize How many candidate sequences the model considers simultaneously.\\
        \keyword{Wider beam = more thorough search} but slower.\\
        Tested: 5 $\rightarrow$ 50 candidates.}
      \end{block}

      \vspace{2pt}

      \begin{block}{\small Length Penalty}
        {\scriptsize Bias toward longer or shorter output.\\
        Negative = encourages brevity (risk: empty output).\\
        Positive = encourages length (risk: hallucination).\\
        Tested: $-$0.5 $\rightarrow$ 2.0.}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{\small Temperature}
        {\scriptsize Controls randomness in word selection.\\
        0 = always pick most likely word.\\
        $>$1 = more creative/random.\\
        Tested: 0.3 $\rightarrow$ 1.5.}
      \end{block}

      \vspace{2pt}

      \begin{block}{\small Sampling vs Greedy}
        {\scriptsize Greedy: always pick best word.\\
        Nucleus: sample from top candidates.\\
        Trade-off: diversity vs reliability.}
      \end{block}
    \end{column}
  \end{columns}

  \vspace{6pt}
  \centering
  {\textcolor{ArgosTeal}{\textit{``Like adjusting a stereo's settings --- you can change the sound,\\but you cannot change the recording.''}}}

  \note{
    Brief primer for non-specialists before the tuning experiments.
    Four key parameters: beam size (search breadth), length penalty (output length bias),
    temperature (randomness), and sampling strategy (greedy vs probabilistic).
    The stereo analogy captures the key insight: these parameters cannot fix fundamental
    visual encoder limitations, only adjust how the LLM uses the visual information.
  }
\end{frame}

    PC1 (68.4\%): Signal Quality --- all 5 content signals load equally (0.43--0.47).

    WER is UNRELIABLE across configs: correlation swings from -0.95 to -0.45.
