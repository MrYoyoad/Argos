\begin{frame}[shrink=8]{Why Engineering Was Hard}
  \twocol{0.50}{
    \alertmetric{85+}{Engineering Challenges}

    \vspace{2pt}

    {\scriptsize\textbf{Research code $\rightarrow$ production:}}
    \begin{itemize}\scriptsize
      \item Container bugs: 37\\
            {\tiny Installation, deployment, GPU, runtime}
      \item EC2$\leftrightarrow$container sync: 26 items\\
            {\tiny Hardcoded paths, venv mismatches, mounted FS}
      \item Pipeline stages broken: 6/8\\
            {\tiny Missing files, wrong venvs, offline failures}
      \item Critical infra fixes: 12\\
            {\tiny NVENC corruption, CUDA OOM, desktop launcher}
    \end{itemize}
  }{0.47}{
    \begin{alertblock}{\small Critical Bug: NVENC Corruption}
      {\scriptsize GPU video encoder silently destroyed\\
      \alert{43\% of output videos}\\[2pt]
      {\tiny Root cause: bash file descriptors leaked\\
      into FFmpeg subprocess, corrupting stream.\\
      Fix: FD isolation + CPU fallback path.}}
    \end{alertblock}

    \vspace{2pt}

    \begin{exampleblock}{\scriptsize Container Pain Points}
      {\tiny No internet (Whisper fails), 12GB vs 24GB GPU,\\
      \texttt{/home/ubuntu} vs \texttt{/workspace} paths,\\
      5 desktop launcher rewrites, venv conflicts}
    \end{exampleblock}
  }

  \note{
    85+ total engineering challenges, not just 37 bugs.
    Pipeline was broken at 6 of 8 stages when containerized.
    26 pending EC2-to-container sync items.
    Container had no internet, different GPU, different paths, mounted FS restrictions.
    Desktop launcher alone needed 5 separate fixes to work.
    NVENC corruption: 43\% of videos destroyed silently.
  }
\end{frame}

\begin{frame}[shrink=5]{Modular Refactoring}
  \twocol{0.47}{

      Automatic venv management\\
      Standardized logging \& error handling\\
      Each module independently testable}

  }

  \vspace{4pt}

  \begin{columns}[T]
    \begin{column}{0.23\textwidth}
      \centering
      \metricbox{52\%}{Line Reduction}
    \end{column}
    \begin{column}{0.23\textwidth}
      \centering
      \metricbox{11}{Modules}
    \end{column}
    \begin{column}{0.23\textwidth}
      \centering
      \successmetric{37}{Tests}
    \end{column}
    \begin{column}{0.23\textwidth}
      \centering
      \metricbox{2}{Environments}
    \end{column}
  \end{columns}

  \note{
    Pipeline was fundamentally broken in the container environment.
    6 of 8 stages had blocking bugs: missing files, wrong venvs, hardcoded paths.
    Refactored from 823 to 393 lines with environment-aware modules.
    37 tests validate both EC2 and container configurations.
    Automatic EC2 vs container detection eliminates path issues.
  }
