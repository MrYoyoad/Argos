      \item Deployment \& GPU: 12 bugs\\
            {\scriptsize NVENC encoding, Docker GPU pass-through}
      \item Runtime \& inference: 12 bugs\\
            {\scriptsize Upload handling, terminal state, metrics}
    \end{itemize}
  }{0.47}{
    \begin{alertblock}{\small Critical Bug: NVENC Corruption}
      GPU video encoder silently destroyed\\
      \alert{43\% of output videos}\\[3pt]
      {\scriptsize Root cause: bash file descriptors leaked\\
      into FFmpeg subprocess, corrupting stream.\\
      Fix: FD isolation + CPU fallback path.}
    \end{alertblock}

    \vspace{4pt}

    \begin{exampleblock}{\scriptsize Other Challenges}
      {\scriptsize HDR tone mapping, Cython builds,\\
      fairseq patches, venv isolation}
    \end{exampleblock}
  }

  \note{
    37 bugs across 3 categories: install, deploy, runtime.
    NVENC corruption is the headline: 43\% of videos destroyed silently.
    Root cause was bash file descriptors leaking into FFmpeg subprocess.
    This is why engineering effort was substantial --- research code is not production code.
  }
\end{frame}

\begin{frame}{Smart Features}
  \vspace{2pt}
  \begin{columns}[T]
    \begin{column}{0.31\textwidth}
      \begin{block}{\small \faIcon{redo}~Transcription Reuse}
        {\scriptsize Manual transcriptions persist\\
        across pipeline runs.\\[3pt]
        Whisper skips matched segments,\\
        saving hours of processing.\\[3pt]
        \texttt{.transcriptions/} directory\\
        preserves all corrections.}
      \end{block}
    \end{column}
    \begin{column}{0.31\textwidth}
      \begin{block}{\small \faIcon{bullseye}~Golden K-Means}
        {\scriptsize Pre-computed cluster centroids\\
        from 1,396-video baseline.\\[3pt]
        Ensures consistent clustering\\
        across all pipeline runs.\\[3pt]
        Eliminates randomness in\\
        feature quantization.}
      \end{block}
    \end{column}
    \begin{column}{0.31\textwidth}
      \begin{block}{\small \faIcon{cut}~Video Segmentation}
        {\scriptsize Automatic splitting for\\
        long videos ($>$30s).\\[3pt]
        Overlap handling between\\
        adjacent segments.\\[3pt]
        Preserves sentence boundaries\\
        where possible.}
      \end{block}
    \end{column}
  \end{columns}

  \note{
    Three production features beyond basic pipeline:
    1. Transcription reuse: manual corrections persist, Whisper skips.
    2. Golden k-means: consistent baseline from 1,396-video model.
    3. Segmentation: handles long videos with overlap.
  }
\end{frame}
