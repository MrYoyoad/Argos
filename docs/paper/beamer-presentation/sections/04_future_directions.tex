% What We Learned: Key Research Insights
% ----------------------------------------------------------
\begin{frame}[shrink=5]{What We Learned: Key Research Insights}
  \vspace{2pt}
  {\small Five findings that directly inform the improvement roadmap:}

  \vspace{6pt}

  \begin{enumerate}
    \item \keyword{WER dramatically overstates failure}\\
          {\scriptsize IS reveals 4$\times$ more useful output (40\% vs 11\%). LLM Judge confirms: 65\% Y+P.}
    \item \keyword{The visual encoder is the primary bottleneck}\\
          {\scriptsize Segment rankings stable across all 16 decode configs (r $>$ 0.92).}
    \item \keyword{Data scarcity limits fine-tuning}\\
          {\scriptsize 1,273 segments is below the $\sim$1K minimum for LoRA generalization (ICLR 2024).}
    \item \keyword{Decode parameter tuning has diminishing returns}\\
          {\scriptsize Best config gained only +0.08 IS. 13 experiments, no breakthrough.}
    \item \keyword{Domain vocabulary confusion is systematic}\\
          {\scriptsize $\sim$284 segments (19\%) show confusion --- requires topic-aware fine-tuning, not just prompt injection.}
  \end{enumerate}

  \vspace{4pt}
  \centering
  \notetext{Each insight maps to a specific roadmap phase.}

  \note{
    Transition slide summarizing the five key findings before presenting the roadmap.
    Each finding directly motivates a specific improvement phase:
    1. WER overstates -> need IS-based confidence scoring.
    2. Visual encoder bottleneck -> need data scaling and encoder improvements.
    3. Data scarcity -> need 20K-50K segments from AVSpeech.
    4. Tuning limits -> don't invest more in decode parameters.
    5. Domain confusion -> topic context injection at decode time.
  }
\end{frame}

\begin{frame}{The LLM: Context Engine for Visual Ambiguity}
  \twocol{0.52}{
    \textbf{\textcolor{ArgosTeal}{What the LLM Does}}

    \vspace{4pt}

    \begin{itemize}\small
      \item Visual encoder outputs a \keyword{probability distribution}\\over phonemes/visemes
      \item Multiple words match the same lip movements\\
            {\scriptsize (admiral/animal, mom/bomb, pat/bat)}
      \item The LLM's job: \textbf{disambiguate}\\using linguistic context
      \item Better LLM = better disambiguation\\= fewer hallucinations, better entities
    \end{itemize}
  }{0.45}{
    \begin{alertblock}{\small Current: Llama-2-7B (2023)}
      \begin{itemize}\scriptsize
        \item 32K vocabulary
        \item 4K context window
        \item Trained on web text, not speech
        \item Limited world knowledge
      \end{itemize}
    \end{alertblock}

    \vspace{4pt}

    \begin{exampleblock}{\small Upgrade: Llama 3.1 8B (2024)}
      \begin{itemize}\scriptsize
        \item 128K vocabulary (4$\times$)
        \item 128K context (32$\times$)
        \item $\approx$ Llama-2-70B quality
        \item \keyword{Same hidden\_size (4096) = drop-in}
        \item Setup: \textbf{1--2 hours}
      \end{itemize}
    \end{exampleblock}
  }

  \note{
    Why the LLM matters: it disambiguates visual ambiguity using linguistic context.
    Current Llama-2-7B has limited vocabulary and world knowledge.
    Llama 3.1 8B is a drop-in replacement (same hidden dimension 4096).
    4x vocabulary, 32x context window, quality approaching Llama-2-70B.
    Setup requires only 1-2 hours: swap model files, retrain projection layer.
  }
\end{frame}

\begin{frame}[shrink=5]{Phase 1: Confidence Scoring}
